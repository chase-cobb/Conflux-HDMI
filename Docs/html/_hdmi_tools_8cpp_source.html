80  bool HdmiTools::Initialize()
81  {
82  HdmiHardwareId hardwareId = GetHardwareId();
83 
84  switch (hardwareId)
85  {
86  case HdmiHardwareId::NO_HW_DETECTED:
87  return false;
88  break;
89  case HdmiHardwareId::XBOXHDMI:
90  m_hdmiInterface = new XboxHDMI::XboxHdmi;
91  m_hdmiInterface->LoadConfig();
92  return true;
93  break;
94  default:
95  break;
96  }
97 
98  return false;
