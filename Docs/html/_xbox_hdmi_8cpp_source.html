62  bool XboxHdmi::GetFirmwareVersion(VersionCode* versionCode)
63  {
64  ULONG smbusRead;
65  uint8_t major, minor, patch;
66 
67  if(HalReadSMBusValue(I2C_HDMI_ADRESS, I2C_FIRMWARE_VERSION + 0, false, &smbusRead) == 0) {
68  major = (uint8_t)smbusRead;
69 
70  HalReadSMBusValue(I2C_HDMI_ADRESS, I2C_FIRMWARE_VERSION + 1, false, &smbusRead);
71  minor = (uint8_t)smbusRead;
72 
73  HalReadSMBusValue(I2C_HDMI_ADRESS, I2C_FIRMWARE_VERSION + 2, false, &smbusRead);
74  patch = (uint8_t)smbusRead;
75 
76  // Firmware 1.0.0 will incorrectly report 0.0.0, so let's fix that.
77  if(major == 0 && minor == 0 && patch == 0) {
78  major = 1;
79  }
80 
81  versionCode->SetVersion(major, minor, patch);
82  return true;
83  }
84  return false;
85  }

92  bool XboxHdmi::LoadConfig()
93  {
94  ULONG widescreenMode;
95  ULONG videoOutMode;
96  ULONG lumaSetting = 0;
97  ULONG cbSetting = 0;
98  ULONG crSetting = 0;
99 
100  // If one read fails it will cause all to fail
101  // TODO : Refactor this so that it's less ugly and lowers the
102  // cost of change.
103  bool readSuccessful = true;
104 
105  readSuccessful = (HalReadSMBusValue(I2C_HDMI_ADRESS, I2C_EEPROM_WIDESCREEN, false,
106  &widescreenMode) == 0);
107 
108  if(readSuccessful)
109  {
110  readSuccessful = (HalReadSMBusValue(I2C_HDMI_ADRESS, I2C_EEPROM_MODE_OUT, false,
111  &videoOutMode) == 0);
112  }
113 
114  if(readSuccessful)
115  {
116  readSuccessful = (HalReadSMBusValue(I2C_HDMI_ADRESS, I2C_EEPROM_ADJUST_LUMA, false,
117  &lumaSetting) == 0);
118  }
119 
120  if(readSuccessful)
121  {
122  readSuccessful = (HalReadSMBusValue(I2C_HDMI_ADRESS, I2C_EEPROM_ADJUST_CB, false,
123  &cbSetting) == 0);
124  }
125 
126  if(readSuccessful)
127  {
128  readSuccessful = (HalReadSMBusValue(I2C_HDMI_ADRESS, I2C_EEPROM_ADJUST_CR, false,
129  &crSetting) == 0);
130  }
131 
132  if(readSuccessful)
133  {
134  //Don't allocate new RangedIntValue if they already exist in the map!!
135  ClearFeatureMap();
136 
137  RangedIntValue* widescreenValue = new RangedIntValue((int8_t)widescreenMode, 0, 2, FEATURE_WIDESCREEN_ADJUST);
138  m_featureValues[SupportedFeatures::WIDESCREEN_ADJUST] = widescreenValue;
139 
140  RangedIntValue* videoOutValue = new RangedIntValue((int8_t)videoOutMode, 0, 1, FEATURE_VIDEO_MODE_ADJUST);
141  m_featureValues[SupportedFeatures::VIDEO_MODE_ADJUST] = videoOutValue;
142 
143  RangedIntValue* lumaValue = new RangedIntValue((int8_t)lumaSetting, -12, 12, FEATURE_LUMA_ADJUST);
144  m_featureValues[SupportedFeatures::LUMA_ADJUST] = lumaValue;
145 
146  RangedIntValue* cbValue = new RangedIntValue((int8_t)cbSetting, -12, 12, FEATURE_CB_ADJUST);
147  m_featureValues[SupportedFeatures::CB_ADJUST] = cbValue;
148 
149  RangedIntValue* crValue = new RangedIntValue((int8_t)crSetting, -12, 12, FEATURE_CR_ADJUST);
150  m_featureValues[SupportedFeatures::CR_ADJUST] = crValue;
151  }
152 
153  return readSuccessful;
154  }

192  bool XboxHdmi::GetFirmwareCompileTime(time_t* compileTime)
193  {
194  bool readSuccessful = true;
195  ULONG compileTimeRaw[4];
196 
197  readSuccessful = HalReadSMBusValue(I2C_HDMI_ADRESS, I2C_COMPILE_TIME0, false,
198  &compileTimeRaw[0]) == 0;
199 
200  if(readSuccessful)
201  {
202  readSuccessful = HalReadSMBusValue(I2C_HDMI_ADRESS, I2C_COMPILE_TIME1, false,
203  &compileTimeRaw[1]) == 0;
204  }
205 
206  if(readSuccessful)
207  {
208  readSuccessful = HalReadSMBusValue(I2C_HDMI_ADRESS, I2C_COMPILE_TIME2, false,
209  &compileTimeRaw[2]) == 0;
210  }
211 
212  if(readSuccessful)
213  {
214  readSuccessful = HalReadSMBusValue(I2C_HDMI_ADRESS, I2C_COMPILE_TIME3, false,
215  &compileTimeRaw[3]) == 0;
216  }
217 
218  if(readSuccessful)
219  {
220  *compileTime = (compileTimeRaw[0] << 24) +
221  (compileTimeRaw[1] << 16) +
222  (compileTimeRaw[2] << 8) +
223  (compileTimeRaw[3]);
224  }
225 
226  return readSuccessful;
